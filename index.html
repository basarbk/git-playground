Git Cheat Sheet

Config
git config - configure git system wide, global or local
  • git config {storage} {config.key} {value} # storage is --system, --global or --local
  • git config --global user.name "Your username" # user.name must be set
  • git config --global user.email your@mail.com # user.email must be set
  • git config --global core.editor "code --wait" # setting editor as visual studio code
  • git config --global alias.lg "log --pretty=format:'%C(red)%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(dim magenta)- %ae%Creset' --branches --graph" # custom format log command with lg alias
  • git config --global --unset {config.key} # delete the previously set config
Create
git init - create an empty git repository in current directory
git clone - clone remote repository
  • git clone https://remote-url/repository.git # copies remote into local folder, repository
  • git clone https://remote-url/repository.git folderX # copies remote into local folder, folderX
Branching
git branch - creating/listing/deleting branches
  • git branch # list existing branches
  • git branch branchX # create branch with name branchX
  • git branch branchX {pointer} # pointer is HEAD or the commit id for creating the branchX from specific commit
  • git branch -d branchX # delete branch
  • git branch -D branchX # force delete
  • git branch -m old-branchX new-branchX # rename
  • git branch --merged # list the branches merged to current one
  • git branch --no-merged # list unmerged branches
  • git branch -u originX/branchX # set remote branc as upstream for current branch
  • git branch -a # branch list to include remote branches also
git checkout - switch between branches
  • git checkout branchX # switch to existing branchX
  • git checkout -b branchX # create and switch to branch
  • git checkout -b branchX {pointer} # create a branch from specific commit
git switch - same as checkout
  • git switch -c branchX # create and switch to branch
  • git switch -c branchX {pointer} # create a branch from specific commit
  • git switch - # switch back to previous branch
git stash - stack for moving uncommitted changes to temporary storage
  • git stash push
  • git stash list
  • git stash apply {stash-index} # applies changes in stash. If stash-index (e.g stash@{1}) is not provided, applies the top of the list
  • git stash drop {stash-index} # removes the stashed change. If stash-index (e.g stash@{1}) is not provided, drops the top of the list
  • git stash pop {stash-index} # stash apply & stash drop combined
  • git stash show {stash-index} -p # show the diff in stashed change
  • git stash clear
git merge - applies the commits of the target branch on top of the current branch's head
  • git merge branchX # merge and commit the changes in branchX to current branch
  • git merge --no-commit branchX # merge the branchX to current branch without commit
  • git merge --abort # rollback to previous state from merge conflict
git rebase - applies the commits of this branch on top of the target branch's head
  • git rebase branchX # commit the changes of current branch on top of the branchX last change
  • git rebase --continue # complete the rebase operation after resolving conflict
git cherry-pick - copy commit from one branch to another
  • git cherry-pick {commit-id} # copy the specified commit to current branch
  • git cherry-pick {commit-X}..{commit-Y} # copy the range of commits from the commit after commit-x until the commit-y inclusive
git tag - set a custom name to specific commit
  • git tag -a vX.0 -m "Tag message" # create annotated tag by setting message about it
  • git tag vX.0 # create lightweight tag
  • git tag vX.0 {commit-id} # create tag for specified commit
  • git tag -d vX.0 # delete tag
File Operations
git clean - remove untracked files
  • git clean --dry-run / -n # shows which files would have been removed
  • git clean -f fileX # remove the fileX
  • git clean -d -f # remove all untracked files and directories
git rm - remove tracked files
  • git rm fileX # remove file from staging area and working directory
  • git rm --cached fileX # remove file from staging area
git mv - move/rename tracked files
  • git mv old-fileX new-fileX # rename or move file
Workflow
git add - add file to staging area
  • git add fileX # add fileX to staging area
  • git add folderX # add all the files under folderX to staging area
  • git add . # add all the files to staging area
git commit - copy the snapshot from staged area to local repository
  • git commit # opens editor to set commit message
  • git commit -v # shows the diff in editor
  • git commit -a -m "Commit message" # add all tracked files to staging area and set commit message inline
  • git commit --amend # update last commit
git reset - unstage changes and undoing commit
  • git reset fileX - unstage fileX
  • git reset --hard - reset local working tree and staging area to last commit
  • git reset --soft {pointer} # pointer is HEAD or the commit id. reset to specified point but keep changes in staging area and working directory
  • git reset --mixed {pointer} # --mixed is optional. reset to specified point but keep changes only in working directory
  • git reset --hard {pointer} # reset to specified point and loose all changes after that point
git restore - unstage a file or reset file in working directory
  • git restore fileX - remove changes from specified file in working area
  • git restore --staged fileX - remove changes from specified file in staging area
git checkout - reset file in working directory
  • git checkout fileX
git revert - revert specified commit
  • git revert HEAD # add a new revert commit for the last committed changes
  • git revert {pointer} # pointer is HEAD or the commit id. revert the specified point
Inspect & History
git status - show staged and unstaged files
  • git status
  • git status --short / -s
git diff - show line by line diff
  • git diff # diff between working directory and staging area
  • git diff --staged / --cached # diff between staging area and local repository
git log - lists commit history
  • git log
  • git log --patch / -p # show changes in each commit
  • git log -1 # show only specified number of commits
  • git log --pretty=oneline # print single line of commit logs
  • git log --pretty=format:"%h - %s - (%cr) - %ae" # print logs in custom format
  • git log --branches --graph # print logs from all branches with commit tree graph
git show - shows the commit and diff
  • git show {pointer} # pointer is HEAD or the commit id. If not provided, by default it is HEAD. Shows the commit info and changes
git ls-files - list files in repository
Remote
git remote - remote repository
  • git remote # list remote repositories
  • git remote -v # show the url of remote repositories
  • git remote rename old-originX new-originX # rename remote repository name
  • git remote remove originX # remove the remote repository
git push - pushing local changes to remote
  • git push -u originX branchX # push local branchX to remote, the originX
  • git push originX --delete branchX # delete remote branch
  • git push -f # force push
  • git push originX tagX # push local tagX to remote
  • git push originX --tags # push all local tags
  • git push originX --delete tagX # delete remote tag
git fetch - getting remote changes to local repo
  • git fetch originX # get the latest changes from remote repository. adding remote name is optional if the upstream is set for current branch
  • git fetch -p originX # prune, to apply the remote delete operations locally
  • git fetch -p --prune-tags # delete local tags if they are removed at remote
